import BrowserOnly from '@docusaurus/BrowserOnly';
import LoadingSpinner from '@site/src/components/common/LoadingSpinner';

# Daily Active Users

<BrowserOnly fallback={<LoadingSpinner />}>
  {() => {
    const UsageTimeSeriesChart = require('@site/src/components/Usage/UsageTimeSeriesChart').default;
    const UsageTopAddresses = require('@site/src/components/Usage/UsageTopAddresses').default;
    const { useUsageMetrics } = require('@site/src/hooks/useUsageMetrics');

    function DailyActiveUsersPage() {
      const { data, loading, error } = useUsageMetrics();

      if (loading) {
        return <LoadingSpinner />;
      }

      if (error || !data) {
        return (
          <div className="alert alert--danger" role="alert">
            <strong>Could not load usage metrics.</strong> {error || 'Unknown error'}
          </div>
        );
      }

      const { daily_users } = data;

      return (
        <>
          <p>
            Daily active users are wallets executing fee-generating adjustments such as liquidity changes or collateral
            management. These actions directly produce the revenue captured by the treasury.
          </p>
          <UsageTimeSeriesChart
            title="Daily Active Users"
            data={daily_users.daily_counts}
            yAxisLabel="Wallets"
            description="Unique wallets triggering fee-generating instructions per UTC day."
          />
          <UsageTopAddresses
            title="Top Daily Active Wallets"
            rows={daily_users.top_wallets}
            description="Ranked by the number of distinct days each wallet adjusted positions."
          />
        </>
      );
    }

    return <DailyActiveUsersPage />;
  }}
</BrowserOnly>
