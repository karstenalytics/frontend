    function RandomChartButton() {
      const [destination, setDestination] = useState(null);

      const [topMargin, setTopMargin] = useState('-10rem');

        '/analysis/defituna/revenue-breakdown/by-type',
        '/analysis/defituna/revenue-breakdown/by-pool',
        '/analysis/defituna/revenue-breakdown/pools-vs-types',

        '/analysis/defituna/revenue-breakdown/tx-type-development',
        '/analysis/defituna/revenue-breakdown/orca-vs-fusion',
        '/analysis/defituna/usage-statistics/wallet-usage',
        '/analysis/defituna/usage-statistics/active-stakers',
        '/analysis/defituna/staking/staked-tuna',
        '/analysis/defituna/staking/staker-conviction',
        '/analysis/defituna/staking/vesting-timeline',
        '/analysis/defituna/staking/wallet-timeline',
        '/analysis/defituna/staking-apr',
      ];

      const buttonTexts = [
        "Enough with the intro, bring me to one of your charts already!",
        "Skip the philosophy, show me the data!",
        "TL;DR - Just show me a random chart!",
        "I'm here for the charts, not the story!"
      ];

      useEffect(() => {
        const randomChart = chartUrls[Math.floor(Math.random() * chartUrls.length)];
        const randomText = buttonTexts[Math.floor(Math.random() * buttonTexts.length)];
        setDestination(randomChart);
        setButtonText(randomText);

        const updateMargin = () => {
          const width = window.innerWidth;
          if (width < 577) {
            setTopMargin('-9rem');  // Mobile

            setTopMargin('-10.3rem');  // Tablet

          }

        updateMargin();
        window.addEventListener('resize', updateMargin);
        return () => window.removeEventListener('resize', updateMargin);
      }, []);

      if (!destination) return null;

      return (
        <a
          href={destination}
          style={{
            display: 'block',
            textAlign: 'center',
            padding: '5px 2px 2px 2px',

  {() => {
    const { useEffect, useState } = require('react');
    const useBaseUrl = require('@docusaurus/useBaseUrl').default;

    function StatsCards() {
      const baseUrl = useBaseUrl('/');
      const [txCount, setTxCount] = useState(null);
      const [loading, setLoading] = useState(true);

      useEffect(() => {
        fetch(`${baseUrl}data/usage_metrics.json`)
          .then(response => response.json())
          .then(data => {
            setTxCount(data.summary.transactions_scanned);
            setLoading(false);
          })
          .catch(err => {
            console.error('Error loading transaction count:', err);
            setLoading(false);
          });
      }, [baseUrl]);

      if (loading) {
        return (
          <div style={{
            padding: '24px',
            textAlign: 'center',
            color: 'var(--ifm-color-secondary)',
          }}>
            Loading statistics...
          </div>
        );
      }

      const cardStyle = {
        background: 'var(--ifm-background-surface-color)',
        border: '1px solid var(--ifm-toc-border-color)',
        borderRadius: 'var(--ifm-global-radius)',
        padding: '24px',
        boxShadow: '0 1px 2px 0 rgba(0, 0, 0, 0.05)',
        textAlign: 'center',
        display: 'flex',
        flexDirection: 'column',
        gap: '8px',
      };

      const labelStyle = {
        fontSize: '14px',
        fontWeight: '600',
        color: 'var(--ifm-color-secondary)',
        textTransform: 'uppercase',
        letterSpacing: '0.5px',
      };

      const valueStyle = {
        fontSize: '32px',
        fontWeight: '700',
        color: 'var(--ifm-color-primary)',
        lineHeight: '1.2',
      };

        fontSize: '13px',
        color: 'var(--ifm-color-emphasis-600)',
      };

      return (
        <div style={{

          marginBottom: '48px',
